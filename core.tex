\section{Syntax}

$e ::=$

$| x$

$| \lam{b}{e} | \app{e}{e} | \arr{t}{e}$

$| \elet{b}{e}{e}$

$| \Type{n}{e} | \Level{n} | \lzero | \lsucc{e}$
\\\\
$b ::=$

$| x$

$| [x : e]$
\\\\
$t ::=$

$| [x : e]$

$| e$
